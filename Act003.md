EJERCICIO 1:Encontrar el último elemento de una lista.

```c#
% ===============================================
% Autor: Diego Samuel Garcia Bagnis 
% Fecha: 17 de octubre de 2024
% Descripción: Programa en C# que determina 
%              el último elemento de una lista.
%              Se incluye una versión comentada
%              en C# para referncia. 
% ===============================================
%------------------------------------------------------------------------------
using System;
using System.Collections.Generic;

class Program
{
    /// <summary>
    /// Función para encontrar el último elemento de una lista en C#.
    /// </summary>
    /// <typeparam name="T">El tipo de elementos de la lista.</typeparam>
    /// <param name="list">La lista de la cual se quiere obtener el último elemento.
    /// La lista debe tener al menos un elemento.</param>
    /// <returns>El último elemento de la lista.</returns>
    /// <exception cref="ArgumentException">Lanza una excepción si la lista está vacía.</exception>
    static T MyLast<T>(List<T> list)
    {
        if (list == null || list.Count == 0)
            throw new ArgumentException("La lista no puede estar vacía.");
        
        return list[list.Count - 1];  // Devuelve el último elemento.
    }

    static void Main()
    {
        List<char> lista = new List<char> { 'a', 'b', 'c', 'd' };
        char ultimoElemento = MyLast(lista);
        
        Console.WriteLine($"El último elemento es: {ultimoElemento}");
    }
}

%------------------------------------------------------------------------------
PROGRAMA REPRESENTADO EN PROLOG:
% Encuentra el último elemento de una lista.
% La lista debe tener al menos un elemento.
my_last(X, [X]).
my_last(X, [_|T]) :- my_last(X, T).
```

EJERCICIO 2: Encontrar el penúltimo elemento de una lista.

```c#
% ===============================================
% Autor: Diego Samuel Garcia Bagnis 
% Fecha: 17 de octubre de 2024
% Descripción: Programa en C# que determina 
%              el peúltimo elemento de una lista.
%              Se incluye una versión comentada
%              en C# para referncia. 
% ===============================================
%------------------------------------------------------------------------------
using System;
using System.Collections.Generic;

class Program
{
    /// <summary>
    /// Función para encontrar el penúltimo elemento de una lista en C#.
    /// </summary>
    /// <typeparam name="T">El tipo de elementos de la lista.</typeparam>
    /// <param name="list">La lista de la cual se quiere obtener el penúltimo elemento.
    /// La lista debe tener al menos dos elementos.</param>
    /// <returns>El penúltimo elemento de la lista.</returns>
    /// <exception cref="ArgumentException">Lanza una excepción si la lista tiene menos de dos elementos.</exception>
    static T Penultimate<T>(List<T> list)
    {
        if (list == null || list.Count < 2)
            throw new ArgumentException("La lista debe tener al menos dos elementos.");
        
        return list[list.Count - 2];  // Devuelve el penúltimo elemento.
    }

    static void Main()
    {
        List<char> lista = new List<char> { 'a', 'b', 'c', 'd' };
        char penultimoElemento = Penultimate(lista);
        
        Console.WriteLine($"El penúltimo elemento es: {penultimoElemento}");
    }
}

%------------------------------------------------------------------------------
Aqui esta su variante en PROLOG:
% Encuentra el penúltimo elemento de una lista.
% La lista debe tener al menos dos elementos.
penultimate(X, [X, _]).
penultimate(X, [_|T]) :- penultimate(X, T).
```

EJERCICIO 3: Encontrar el k-ésimo elemento de una lista.
 
```c#
% ===============================================
% Autor: Diego Samuel Garcia Bagnis 
% Fecha: 17 de octubre de 2024
% Descripción: Programa en C# que encuentra el K-esimo en una lista.
%              Se incluye una versión comentada en C# para referncia.
% ===============================================
%------------------------------------------------------------------------------
using System;
using System.Collections.Generic;

class Program
{
    /// <summary>
    /// Función para encontrar el k-ésimo elemento de una lista en C#.
    /// </summary>
    /// <typeparam name="T">El tipo de elementos de la lista.</typeparam>
    /// <param name="list">La lista de la cual se quiere obtener el k-ésimo elemento.</param>
    /// <param name="k">El índice del elemento que se desea (debe ser mayor que 0).</param>
    /// <returns>El k-ésimo elemento de la lista.</returns>
    /// <exception cref="ArgumentException">Lanza una excepción si el índice es menor que 1 
    /// o si el índice es mayor que el número de elementos en la lista.</exception>
    static T ElementAt<T>(List<T> list, int k)
    {
        if (k < 1 || k > list.Count)
            throw new ArgumentException("El índice debe ser mayor que 0 y menor o igual al número de elementos de la lista.");
        
        return list[k - 1];  // Devuelve el k-ésimo elemento (el índice en C# es 0 basado).
    }

    static void Main()
    {
        List<char> lista = new List<char> { 'a', 'b', 'c', 'd' };
        int indice = 3;
        char elemento = ElementAt(lista, indice);
        
        Console.WriteLine($"El elemento en el índice {indice} es: {elemento}");
    }
}
%------------------------------------------------------------------------------
Aqui esta su variante en PROLOG: 
% Encuentra el k-ésimo elemento de una lista.
% El índice K debe ser mayor que 0.
element_at(X, [X|_], 1).
element_at(X, [_|T], K) :- K > 1, K1 is K - 1, element_at(X, T, K1).
```

EJERCICIO 4: Encontrar el número de elementos de una lista.
```c#
% ===============================================
% Autor: Diego Samuel Garcia Bagnis 
% Fecha: 20 de octubre de 2024
% Descripción: Programa en C# que encuentra el número de elementos de una lista
%              Se incluye una versión comentada en C# para referencia.
% ===============================================
%------------------------------------------------------------------------------
using System;

class Program
{
    // Calcula el número de elementos en una lista.
    // Caso base: Si la lista está vacía, la longitud es 0.
    static int ListLength(int[] list)
    {
        // Si la lista no tiene elementos, retorna 0
        if (list.Length == 0)
            return 0;
        
        // Caso recursivo: la longitud es 1 más que la longitud de la lista sin el primer elemento.
        int[] tail = new int[list.Length - 1];
        Array.Copy(list, 1, tail, 0, list.Length - 1);
        return 1 + ListLength(tail);
    }

    static void Main()
    {
        int[] list = {1, 2, 3, 4}; // Lista de ejemplo
        int length = ListLength(list);
        Console.WriteLine("La longitud de la lista es: " + length);
    }
}

%------------------------------------------------------------------------------
Aqui esta su variante en PROLOG: 
% Calcula el número de elementos de una lista.
% Caso base: una lista vacía tiene longitud 0.
list_length([], 0).
% Caso recursivo: la longitud es 1 más que la longitud de la cola.
list_length([_|T], N) :- list_length(T, N1), N is N1 + 1.
```
